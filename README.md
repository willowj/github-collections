      > import sys

      >  
      >     def __getattr__(self, attr):
      >         path = _lazy_modules[self.name]
      >         f, pathname, desc = imp.find_module(self.name, path)

      >         lf = sys.meta_path.pop()

      >  
      >     def load_module(self, name):
